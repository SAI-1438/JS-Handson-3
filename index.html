Q7. Check whether a Number is a prime or not

const Prime_Check = (n) => 
        {
            let factor =0;
             for(i =1;i<=n;i++){
               if(n%i===0){
                 factor++;
               }
             }
            if(factor===2){
              return "YES";
            }
            else{
              return "NO";
            } 
        };
